```java
package com.zbkj.service.dao;

import com.baomidou.mybatisplus.core.mapper.BaseMapper;
import com.zbkj.service.model.TechnicianSchedule;
import org.apache.ibatis.annotations.Param;
import org.apache.ibatis.annotations.Select;
import java.util.List;

public interface TechnicianScheduleMapper extends BaseMapper<TechnicianSchedule> {
    
    /**
     * 查询技师某日排班
     */
    @Select("SELECT * FROM eb_technician_schedule " +
            "WHERE technician_id = #{technicianId} " +
            "AND work_date = #{workDate}")
    TechnicianSchedule getByTechnicianAndDate(
        @Param("technicianId") Integer technicianId,
        @Param("workDate") String workDate
    );
    
    /**
     * 查询门店某日所有排班
     */
    @Select("SELECT * FROM eb_technician_schedule " +
            "WHERE store_id = #{storeId} " +
            "AND work_date = #{workDate}")
    List<TechnicianSchedule> getByStoreAndDate(
        @Param("storeId") Integer storeId,
        @Param("workDate") String workDate
    );
    
    /**
     * 批量插入排班
     */
    int batchInsert(@Param("schedules") List<TechnicianSchedule> schedules);
}
```

## 📅 Day 3-4: 后端API开发

### 核心API列表

#### 1. 智能排班API
```
POST /api/admin/schedule/smart-create
功能：根据历史数据智能生成一周排班
参数：
{
  "storeId": 1,
  "startDate": "2026-02-17",
  "endDate": "2026-02-23",
  "technicianIds": [1, 2, 3, 4],
  "serviceSkuId": 1
}
```

#### 2. 手动创建排班API
```
POST /api/admin/schedule/create
功能：店长手动创建单日排班
参数：
{
  "storeId": 1,
  "technicianId": 1,
  "workDate": "2026-02-17",
  "workStartTime": "09:00",
  "workEndTime": "22:00",
  "serviceSkuId": 1,
  "isOffpeakEnabled": 1
}
```

#### 3. 查询排班API
```
GET /api/admin/schedule/list
功能：查询排班列表（周视图/月视图）
参数：
- storeId: 门店ID
- startDate: 开始日期
- endDate: 结束日期
- technicianId: 技师ID（可选）
```

#### 4. 更新时间槽API
```
PUT /api/admin/schedule/update-slots
功能：更新排班的时间槽状态
参数：
{
  "scheduleId": 1,
  "timeSlots": "[{...}]"
}
```

#### 5. 用户查询可预约时间槽API
```
GET /api/front/schedule/available-slots
功能：用户查询可预约的时间槽
参数：
- storeId: 门店ID
- date: 日期
- technicianId: 技师ID（可选，点钟时必填）
- serviceSkuId: 服务SKU ID
```

```vue
<template>
  <view class="booking-page">
    <!-- 选择日期 -->
    <view class="date-picker">
      <scroll-view scroll-x>
        <view 
          v-for="date in dates" 
          :key="date"
          @click="selectDate(date)"
          :class="['date-item', {active: selectedDate === date}]"
        >
          {{ formatDate(date) }}
        </view>
      </scroll-view>
    </view>
    
    <!-- 选择技师 -->
    <view class="technician-list">
      <view 
        v-for="tech in technicians" 
        :key="tech.id"
        @click="selectTechnician(tech.id)"
        :class="['tech-item', {active: selectedTechId === tech.id}]"
      >
        <image :src="tech.avatar" />
        <text>{{ tech.name }}</text>
        <text class="rating">⭐{{ tech.rating }}</text>
      </view>
    </view>
    
    <!-- 时间槽列表 -->
    <view class="time-slots" v-if="selectedTechId">
      <view 
        v-for="slot in availableSlots" 
        :key="slot.id"
        @click="selectSlot(slot)"
        :class="['slot-item', slot.status, {offpeak: slot.isOffpeak}]"
      >
        <text class="time">{{ slot.startTime }}</text>
        <text class="price">¥{{ slot.price }}</text>
        <text v-if="slot.isOffpeak" class="tag">闲时</text>
      </view>
    </view>
    
    <!-- 确认按钮 -->
    <button @click="confirmBooking" class="confirm-btn">
      立即预约
    </button>
  </view>
</template>

<script>
export default {
  data() {
    return {
      selectedDate: '',
      selectedTechId: null,
      selectedSlot: null,
      availableSlots: []
    }
  },
  methods: {
    async selectTechnician(techId) {
      this.selectedTechId = techId
      await this.loadAvailableSlots()
    },
    
    async loadAvailableSlots() {
      const res = await this.$api.get('/front/schedule/available-slots', {
        storeId: this.storeId,
        date: this.selectedDate,
        technicianId: this.selectedTechId,
        serviceSkuId: this.serviceSkuId
      })
      this.availableSlots = res.data
    },
    
    async confirmBooking() {
      // 创建预约订单
      const res = await this.$api.post('/front/order/create-booking', {
        scheduleId: this.selectedSlot.scheduleId,
        slotId: this.selectedSlot.id,
        technicianId: this.selectedTechId,
        reserveDate: this.selectedDate,
        reserveTime: this.selectedSlot.startTime
      })
      
      // 跳转支付
      uni.navigateTo({
        url: `/pages/order/pay?orderNo=${res.data.orderNo}`
      })
    }
  }
}
</script>
```

## 🐛 常见问题

**Q1: 时间槽JSON格式是什么？**
```json
[
  {
    "id": "slot_0900",
    "startTime": "09:00",
    "endTime": "10:00",
    "status": "available",
    "price": 128.00,
    "isOffpeak": false,
    "orderNo": ""
  }
]
```

**Q2: 如何判断时间槽是否可预约？**
- status = "available" 且 orderNo 为空

**Q3: 闲时优惠如何设置？**
- 在排班时设置 `is_offpeak_enabled = 1`
- 时间槽JSON中 `isOffpeak = true`
- 价格自动降低（如¥128 → ¥98）
