**CardServiceImpl.java** - 核心实现

```java
package com.zbkj.service.service.impl;

import com.zbkj.service.service.CardService;
import com.zbkj.service.model.*;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

@Service
public class CardServiceImpl implements CardService {
    
    @Autowired
    private UserCardMapper userCardMapper;
    
    @Autowired
    private CardTemplateMapper cardTemplateMapper;
    
    @Autowired
    private CardUsageMapper cardUsageMapper;
    
    /**
     * 购买卡项
     */
    @Override
    @Transactional(rollbackFor = Exception.class)
    public String purchaseCard(Integer userId, Integer cardTemplateId, Integer quantity) {
        // 1. 查询卡项模板
        CardTemplate template = cardTemplateMapper.selectById(cardTemplateId);
        if (template == null || template.getStatus() != 1) {
            throw new RuntimeException("卡项不存在或已下架");
        }
        
        // 2. 创建订单（复用CRMEB订单系统）
        StoreOrder order = new StoreOrder();
        order.setOrderType(3); // 3=卡项订单
        order.setUid(userId);
        order.setTotalPrice(template.getSellPrice().multiply(new BigDecimal(quantity)));
        order.setPayPrice(template.getSellPrice().multiply(new BigDecimal(quantity)));
        // ... 其他订单字段
        
        String orderNo = storeOrderService.create(order);
        
        return orderNo;
    }
    
    /**
     * 支付成功回调 - 生成持卡记录
     */
    public void onPaySuccess(String orderNo) {
        StoreOrder order = storeOrderService.getByOrderNo(orderNo);
        
        if (order.getOrderType() == 3) { // 卡项订单
            // 生成持卡记录
            UserCard userCard = new UserCard();
            userCard.setUserId(order.getUid());
            userCard.setCardTemplateId(order.getProductId());
            userCard.setCardNo(generateCardNo());
            
            CardTemplate template = cardTemplateMapper.selectById(order.getProductId());
            userCard.setCardType(template.getCardType());
            
            if (template.getCardType() == 1) { // 次卡
                userCard.setTotalTimes(template.getTotalTimes());
                userCard.setRemainingTimes(template.getTotalTimes());
                userCard.setUsedTimes(0);
            } else if (template.getCardType() == 3) { // 储值卡
                userCard.setTotalAmount(template.getTotalAmount().add(template.getBonusAmount()));
                userCard.setRemainingAmount(template.getTotalAmount().add(template.getBonusAmount()));
                userCard.setUsedAmount(BigDecimal.ZERO);
            }
            
            userCard.setPurchaseTime((int)(System.currentTimeMillis() / 1000));
            userCard.setExpireTime((int)(System.currentTimeMillis() / 1000) + template.getValidDays() * 86400);
            userCard.setStatus(1);
            userCard.setCreatedAt((int)(System.currentTimeMillis() / 1000));
            
            userCardMapper.insert(userCard);
        }
    }
    
    /**
     * 核销卡项
     */
    @Override
    @Transactional(rollbackFor = Exception.class)
    public Boolean verifyCard(String cardNo, Long orderId, Integer usageType, 
                              Integer usageTimes, BigDecimal usageAmount) {
        // 1. 查询卡项
        UserCard card = userCardMapper.selectOne(
            new LambdaQueryWrapper<UserCard>()
                .eq(UserCard::getCardNo, cardNo)
        );
        
        if (card == null) {
            throw new RuntimeException("卡号不存在");
        }
        
        // 2. 检查卡状态
        if (card.getStatus() != 1) {
            throw new RuntimeException("卡已失效");
        }
        
        // 3. 检查有效期
        int now = (int)(System.currentTimeMillis() / 1000);
        if (card.getExpireTime() > 0 && card.getExpireTime() < now) {
            card.setStatus(3); // 已过期
            userCardMapper.updateById(card);
            throw new RuntimeException("卡已过期");
        }
        
        // 4. 核销次卡
        if (usageType == 1) {
            if (card.getRemainingTimes() < usageTimes) {
                throw new RuntimeException("卡余次不足");
            }
            
            int beforeTimes = card.getRemainingTimes();
            card.setUsedTimes(card.getUsedTimes() + usageTimes);
            card.setRemainingTimes(card.getRemainingTimes() - usageTimes);
            
            if (card.getRemainingTimes() == 0) {
                card.setStatus(2); // 已用完
            }
            
            // 记录核销
            CardUsage usage = new CardUsage();
            usage.setUserCardId(card.getId());
            usage.setUserId(card.getUserId());
            usage.setOrderId(orderId);
            usage.setUsageType(1);
            usage.setUsageTimes(usageTimes);
            usage.setBeforeTimes(beforeTimes);
            usage.setAfterTimes(card.getRemainingTimes());
            usage.setCreatedAt(now);
            
            cardUsageMapper.insert(usage);
        }
        
        // 5. 核销储值卡
        else if (usageType == 2) {
            if (card.getRemainingAmount().compareTo(usageAmount) < 0) {
                throw new RuntimeException("卡余额不足");
            }
            
            BigDecimal beforeAmount = card.getRemainingAmount();
            card.setUsedAmount(card.getUsedAmount().add(usageAmount));
            card.setRemainingAmount(card.getRemainingAmount().subtract(usageAmount));
            
            if (card.getRemainingAmount().compareTo(BigDecimal.ZERO) == 0) {
                card.setStatus(2); // 已用完
            }
            
            // 记录核销
            CardUsage usage = new CardUsage();
            usage.setUserCardId(card.getId());
            usage.setUserId(card.getUserId());
            usage.setOrderId(orderId);
            usage.setUsageType(2);
            usage.setUsageAmount(usageAmount);
            usage.setBeforeAmount(beforeAmount);
            usage.setAfterAmount(card.getRemainingAmount());
            usage.setCreatedAt(now);
            
            cardUsageMapper.insert(usage);
        }
        
        card.setUpdatedAt(now);
        userCardMapper.updateById(card);
        
        return true;
    }
    
    /**
     * 生成卡号
     */
    @Override
    public String generateCardNo() {
        // HXY + 日期 + 6位随机数
        String date = new SimpleDateFormat("yyyyMMdd").format(new Date());
        String random = String.format("%06d", new Random().nextInt(999999));
        return "HXY" + date + random;
    }
}
```

**页面路径**: `uniapp/pages/card/mall.vue`

```vue
<template>
  <view class="card-mall">
    <view class="banner">
      <image src="/static/card-banner.jpg" mode="widthFix" />
    </view>
    
    <!-- 卡项列表 -->
    <view class="card-list">
      <view 
        v-for="card in cardList"

        class="card-item"
        @click="showDetail(card)"
      >
        <view class="card-header">
          <text class="card-name">{{ card.name }}</text>
          <text class="card-tag">{{ getCardTypeText(card.cardType) }}</text>
        </view>
        
        <view class="card-content">
          <text class="card-desc">{{ getCardDesc(card) }}</text>
        </view>
        
        <view class="card-footer">
          <view class="price-box">
            <text class="sell-price">¥{{ card.sellPrice }}</text>
            <text class="market-price">¥{{ card.marketPrice }}</text>
          </view>
          <button class="buy-btn" @click.stop="buyCard(card)">
            立即购买
          </button>
        </view>
      </view>
    </view>
  </view>
</template>

<script>
export default {
  data() {
    return {
      cardList: []
    }
  },
  onLoad() {
    this.loadCardList()
  },
  methods: {
    async loadCardList() {
      const res = await this.$api.get('/front/card/template-list')
      this.cardList = res.data
    },
    
    getCardTypeText(type) {
      const map = {1: '次卡', 2: '期卡', 3: '储值卡'}
      return map[type] || ''
    },
    
    getCardDesc(card) {
      if (card.cardType === 1) {
        return `${card.totalTimes}次服务，单次仅¥${(card.sellPrice / card.totalTimes).toFixed(0)}`
      } else if (card.cardType === 2) {
        return `${card.validDays}天内无限次使用`
      } else {
        return `充¥${card.totalAmount}送¥${card.bonusAmount}`
      }
    },
    
    async buyCard(card) {
      const res = await this.$api.post('/front/card/purchase', {
        cardTemplateId: card.id,
        quantity: 1
      })
      
      // 跳转支付
      uni.navigateTo({
        url: `/pages/order/pay?orderNo=${res.data.orderNo}`
      })
    }
  }
}
</script>
```

```vue
<template>
  <view class="verify-card">

        <text class="value highlight">{{ getRemaining(cardInfo) }}</text>
      </view>
      <view class="info-item">
        <text class="label">状态：</text>
        <text class="value">{{ getStatusText(cardInfo.status) }}</text>
      </view>
    </view>
    
    <!-- 核销按钮 -->
    <button 
      v-if="cardInfo && cardInfo.status === 1"
      @click="verifyCard"
      class="verify-btn"
    >
      确认核销
    </button>
  </view>
</template>

<script>
export default {
  data() {
    return {
      cardNo: '',
      cardInfo: null,
      orderId: null
    }
  },
  onLoad(options) {
    this.orderId = options.orderId
  },
  methods: {
    async checkCard() {
      const res = await this.$api.get('/front/card/detail', {
        cardNo: this.cardNo
      })
      this.cardInfo = res.data
    },
    
    async verifyCard() {
      const res = await this.$api.post('/admin/card/verify', {
        cardNo: this.cardNo,
        orderId: this.orderId,
        usageType: this.cardInfo.cardType === 3 ? 2 : 1,
        usageTimes: 1,
        usageAmount: this.orderAmount
      })
      
      uni.showToast({
        title: '核销成功',
        icon: 'success'
      })
      
      setTimeout(() => {
        uni.navigateBack()
      }, 1500)
    },
    
    scanCode() {
      uni.scanCode({
        success: (res) => {
          this.cardNo = res.result
          this.checkCard()
        }
      })
    }
  }
}
</script>
```
