```bash
# 启动服务
systemctl start nginx php7.4-fpm mysql

# 停止服务
systemctl stop nginx php7.4-fpm mysql

# 重启服务
systemctl restart nginx php7.4-fpm mysql

# 查看状态
systemctl status nginx php7.4-fpm mysql
```

### 日志查看
```bash
# Nginx 访问日志
tail -f /var/log/nginx/hexiaoyue_access.log

# Nginx 错误日志
tail -f /var/log/nginx/hexiaoyue_error.log

# PHP-FPM 日志
tail -f /var/log/php7.4-fpm.log

# MySQL 日志
tail -f /var/log/mysql/error.log
```

### 数据库管理
```bash
# 登录 MySQL
mysql -u root -p

# 备份数据库
mysqldump -u root -p crmeb > /root/crmeb/backup/crmeb_$(date +%Y%m%d).sql

# 恢复数据库
mysql -u root -p crmeb < /root/crmeb/backup/crmeb_YYYYMMDD.sql
```

## 💡 常见问题

### Q1: 访问提示 502 Bad Gateway
```bash
# 检查 PHP-FPM 是否运行
systemctl status php7.4-fpm

# 重启 PHP-FPM
systemctl restart php7.4-fpm
```

### Q2: 访问提示 Permission denied
```bash
# 检查目录权限
ls -la /root/crmeb/CRMEB-5.6.3.1/crmeb/public/

# 修复权限
chmod 755 /root /root/crmeb /root/crmeb/CRMEB-5.6.3.1 /root/crmeb/CRMEB-5.6.3.1/crmeb
chmod -R 777 /root/crmeb/CRMEB-5.6.3.1/crmeb/runtime
chmod -R 777 /root/crmeb/CRMEB-5.6.3.1/crmeb/public
```

### Q3: 数据库连接失败
```bash
# 检查 MySQL 是否运行
systemctl status mysql

# 检查 .env 配置
cat /root/crmeb/CRMEB-5.6.3.1/crmeb/.env | grep -E "HOSTNAME|DATABASE|USERNAME|PASSWORD"

# 测试数据库连接
mysql -u root -p -e "SHOW DATABASES;"
```

### Q4: 接口返回 404
```bash
# 检查路由文件
cat /root/crmeb/CRMEB-5.6.3.1/crmeb/app/api/route/v1.php

# 清除缓存
rm -rf /root/crmeb/CRMEB-5.6.3.1/crmeb/runtime/cache/*

# 重启 PHP-FPM
systemctl restart php7.4-fpm
```
