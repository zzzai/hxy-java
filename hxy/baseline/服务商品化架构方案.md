```java
// 复用CRMEB的StoreOrderController，添加预约相关方法

@RestController
@RequestMapping("/api/front/order")
public class StoreOrderController {
    
    /**
     * 创建服务预约订单
     */
    @PostMapping("/create/service")
    public CommonResult<String> createServiceOrder(@RequestBody ServiceOrderRequest request) {
        // 1. 检查商品是否为服务类型
        StoreProduct product = productService.getById(request.getProductId());
        if (product.getProductType() != 2) {
            return CommonResult.failed("该商品不是服务商品");
        }
        
        // 2. 检查时间槽是否可用
        TechnicianSchedule schedule = scheduleService.getById(request.getScheduleId());
        boolean available = timeSlotService.checkSlotAvailable(
            schedule.getTimeSlots(), 
            request.getSlotId()
        );
        if (!available) {
            return CommonResult.failed("该时间段已被预约");
        }
        
        // 3. 锁定时间槽（使用Redis分布式锁）
        String lockKey = String.format("slot_lock:%d:%s", 
            request.getScheduleId(), request.getSlotId());
        boolean locked = redisUtil.lock(lockKey, 300); // 5分钟锁定
        
        if (!locked) {
            return CommonResult.failed("预约失败，请重试");
        }
        
        try {
            // 4. 创建订单（复用CRMEB订单创建逻辑）
            StoreOrder order = new StoreOrder();
            order.setOrderType(2); // 服务订单
            order.setTechnicianId(request.getTechnicianId());
            order.setReserveDate(request.getReserveDate());
            order.setSlotId(request.getSlotId());
            order.setScheduleId(request.getScheduleId());
            order.setLockedExpire((int)(System.currentTimeMillis() / 1000) + 300);
            
            // 复用CRMEB的订单创建流程
            String orderNo = storeOrderService.create(order);
            
            // 5. 更新时间槽状态为"已锁定"
            timeSlotService.lockSlot(request.getScheduleId(), request.getSlotId(), orderNo);
            
            return CommonResult.success(orderNo);
            
        } finally {
            redisUtil.unlock(lockKey);
        }
    }
    
    /**
     * 支付成功回调（扩展CRMEB原有方法）
     */
    @Override
    public void paySuccessCallback(String orderNo) {
        StoreOrder order = storeOrderService.getByOrderNo(orderNo);
        
        if (order.getOrderType() == 2) {
            // 服务订单：更新时间槽状态为"已预约"
            timeSlotService.confirmSlot(order.getScheduleId(), order.getSlotId());
            
            // 生成核销码
            String checkInCode = generateCheckInCode();
            order.setCheckInCode(checkInCode);
            storeOrderService.updateById(order);
        }
        
        // 调用CRMEB原有的支付成功逻辑
        super.paySuccessCallback(orderNo);
    }
}
```

```vue
<script>
export default {
  methods: {
    generateTimeSlots() {
      // 自动生成时间槽：10:00-22:00，每60分钟一个槽
      const slots = [];
      for (let hour = 10; hour < 22; hour++) {
        slots.push({
          id: `slot_${hour}00`,
          startTime: `${hour}:00`,
          endTime: `${hour + 1}:00`,
          status: 'available',
          statusText: '可预约'
        });
      }
      this.timeSlots = slots;
    },
    
    async saveSchedule() {
      const schedule = {
        technicianId: this.technicianId,
        workDate: this.workDate,
        serviceSkuId: this.serviceSkuId,
        timeSlots: JSON.stringify(this.timeSlots)
      };
      
      await this.$api.post('/admin/schedule/create', schedule);
      this.$message.success('排班保存成功');
    }
  }
}
</script>
```

#### 3.2 后端排班API
```java
@RestController
@RequestMapping("/api/admin/schedule")
public class ScheduleController {
    
    @Autowired
    private TechnicianScheduleMapper scheduleMapper;
    
    /**
     * 创建排班
     */
    @PostMapping("/create")
    public CommonResult<Integer> createSchedule(@RequestBody ScheduleRequest request) {
        TechnicianSchedule schedule = new TechnicianSchedule()
            .setTechnicianId(request.getTechnicianId())
            .setWorkDate(request.getWorkDate())
            .setServiceSkuId(request.getServiceSkuId())
            .setTimeSlots(request.getTimeSlots())
            .setTotalSlots(countSlots(request.getTimeSlots()))
            .setAvailableSlots(countSlots(request.getTimeSlots()))
            .setStatus(1)
            .setCreatedAt((int)(System.currentTimeMillis() / 1000));
        
        scheduleMapper.insert(schedule);
        return CommonResult.success(schedule.getId());
    }
    
    /**
     * 查询排班列表
     */
    @GetMapping("/list")
    public CommonResult<List<TechnicianSchedule>> getScheduleList(
        @RequestParam Integer technicianId,
        @RequestParam String startDate,
        @RequestParam String endDate
    ) {
        LambdaQueryWrapper<TechnicianSchedule> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(TechnicianSchedule::getTechnicianId, technicianId);
        wrapper.between(TechnicianSchedule::getWorkDate, startDate, endDate);
        
        List<TechnicianSchedule> schedules = scheduleMapper.selectList(wrapper);
        return CommonResult.success(schedules);
    }
}
```

#### 4.2 核销API
```java
@RestController
@RequestMapping("/api/admin/checkin")
public class CheckInController {
    
    /**
     * 核销订单
     */
    @PostMapping("/verify")
    public CommonResult<Boolean> verifyOrder(@RequestParam String checkInCode) {
        // 1. 查询订单
        LambdaQueryWrapper<StoreOrder> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(StoreOrder::getCheckInCode, checkInCode);
        StoreOrder order = storeOrderMapper.selectOne(wrapper);
        
        if (order == null) {
            return CommonResult.failed("核销码无效");
        }
        
        if (order.getCheckInTime() > 0) {
            return CommonResult.failed("该订单已核销");
        }
        
        // 2. 更新订单状态
        order.setCheckInTime((int)(System.currentTimeMillis() / 1000));
        order.setStatus(3); // 已核销
        storeOrderMapper.updateById(order);
        
        // 3. 更新时间槽状态
        timeSlotService.completeSlot(order.getScheduleId(), order.getSlotId());
        
        // 4. 增加技师服务单数
        technicianMapper.incrementOrderCount(order.getTechnicianId());
        
        return CommonResult.success(true);
    }
}
```

## 📊 核心数据流

### 预约流程
```
用户选择服务 → 选择门店 → 选择日期 → 选择技师 → 选择时间槽
    ↓
锁定时间槽（Redis锁，5分钟）
    ↓
创建订单（order_type=2）
    ↓
支付成功 → 确认时间槽 → 生成核销码
    ↓
到店核销 → 完成服务 → 更新技师统计
```

### 库存管理
```
服务商品库存 = 每天的可预约时间槽数量
- 技师A：10:00-22:00，每小时一个槽 = 12个库存
- 技师B：14:00-22:00，每小时一个槽 = 8个库存
- 总库存 = 20个

预约时：锁定库存（available_stock - 1, locked_stock + 1）
支付后：扣减库存（locked_stock - 1, sold_stock + 1）
取消时：释放库存（locked_stock - 1, available_stock + 1）
```
