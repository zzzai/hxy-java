## 🔌 接口设计

### 1. 会员信息接口

**接口地址**: `GET /api/user/member_info`

**请求头**:
```
Authorization: Bearer {token}
```

**请求参数**: 无

**返回示例**:
```json
{
  "code": 200,
  "message": "success",
  "data": {
    "uid": 10001,
    "nickname": "微信用户",
    "avatar": "https://xxx.com/avatar.jpg",
    "level": 1,
    "level_name": "V1 荷叶",
    "level_icon": "🌸",
    "growth_value": 1950,
    "next_level_growth": 3000,
    "progress": 65,
    "need_money": 700,
    "need_growth": 1050,
    "vip_status": true,
    "member_days": 243,
    "greeting": "晚上好！又是元气满满的一天",
    "discount": 0.9,
    "discount_text": "全场9折"
  }
}
```

**字段说明**:
| 字段 | 类型 | 说明 |
|------|------|------|
| uid | int | 用户ID |
| nickname | string | 用户昵称 |
| avatar | string | 用户头像 |
| level | int | 会员等级（0=荷芽, 1=荷叶, 2=荷花） |
| level_name | string | 等级名称 |
| level_icon | string | 等级图标 |
| growth_value | int | 当前成长值 |
| next_level_growth | int | 下一等级所需成长值 |
| progress | int | 进度百分比（0-100） |
| need_money | float | 还需消费金额 |
| need_growth | int | 还需成长值 |
| vip_status | bool | 是否VIP |
| member_days | int | 会员天数 |
| greeting | string | 问候语 |
| discount | float | 会员折扣（0.9=9折） |
| discount_text | string | 折扣文案 |

## 💻 后端代码实现

### 文件1: `app/api/controller/user/MemberController.php`

```php
<?php
namespace app\api\controller\user;

use app\Request;
use app\services\user\UserLevelService;

/**
 * 会员相关接口
 */
class MemberController
{
    /**
     * 获取会员信息
     * @param Request $request
     * @return \think\Response
     */
    public function memberInfo(Request $request)
    {
        $uid = $request->uid();
        
        /** @var UserLevelService $service */
        $service = app()->make(UserLevelService::class);
        
        try {
            $memberInfo = $service->getUserMemberInfo($uid);
            return app('json')->success($memberInfo);
        } catch (\Exception $e) {
            return app('json')->fail($e->getMessage());
        }
    }
    
    /**
     * 获取会员权益列表
     * @param Request $request
     * @return \think\Response
     */
    public function benefits(Request $request)
    {
        $uid = $request->uid();
        
        /** @var UserLevelService $service */
        $service = app()->make(UserLevelService::class);
        
        try {
            $benefits = $service->getMemberBenefits($uid);
            return app('json')->success(['benefits' => $benefits]);
        } catch (\Exception $e) {
            return app('json')->fail($e->getMessage());
        }
    }
}
```

### 文件2: `app/services/user/UserLevelService.php` (扩展)

在现有的 `UserLevelService.php` 中添加以下方法：

```php
/**
 * 获取用户会员信息（用于前端展示）
 * @param int $uid
 * @return array
 */
public function getUserMemberInfo(int $uid): array
{
    $user = User::where('uid', $uid)->find();
    if (!$user) {
        throw new \Exception('用户不存在');
    }
    
    $userLevel = UserLevel::where('uid', $uid)->find();
    if (!$userLevel) {
        // 如果没有会员记录，创建默认记录
        $userLevel = $this->initUserLevel($uid);
    }
    
    // 获取等级配置
    $levelConfig = $this->getLevelConfig($userLevel->level);
    
    // 计算进度
    $progress = 0;
    $needGrowth = 0;
    $needMoney = 0;
    
    if ($userLevel->level < 2) { // 未达到最高等级
        $nextLevelConfig = $this->getLevelConfig($userLevel->level + 1);
        $needGrowth = $nextLevelConfig['growth_value'] - $userLevel->growth_value;
        $needMoney = $needGrowth; // 1元=1成长值
        $progress = intval(($userLevel->growth_value / $nextLevelConfig['growth_value']) * 100);
    } else {
        $progress = 100; // 已达到最高等级
    }
    
    // 生成问候语
    $greeting = $this->getGreeting();
    
    return [
        'uid' => $uid,
        'nickname' => $user->nickname,
        'avatar' => $user->avatar,
        'level' => $userLevel->level,
        'level_name' => $levelConfig['name'],
        'level_icon' => $levelConfig['icon'],
        'growth_value' => $userLevel->growth_value,
        'next_level_growth' => $userLevel->level < 2 ? $nextLevelConfig['growth_value'] : $userLevel->growth_value,
        'progress' => $progress,
        'need_money' => $needMoney,
        'need_growth' => $needGrowth,
        'vip_status' => $userLevel->level > 0,
        'member_days' => intval((time() - strtotime($userLevel->create_time)) / 86400),
        'greeting' => $greeting,
        'discount' => $levelConfig['discount'],
        'discount_text' => $levelConfig['discount_text'],
    ];
}

/**
 * 获取会员权益列表
 * @param int $uid
 * @return array
 */
public function getMemberBenefits(int $uid): array
{
    // 获取待评价订单数
    $reviewCount = \app\model\store\StoreOrder::where('uid', $uid)
        ->where('status', 2) // 已完成
        ->where('is_review', 0) // 未评价
        ->count();
    
    return [
        [
            'id' => 1,
            'name' => '荷有礼',
            'sub_name' => '送新意礼',
            'icon' => '/static/images/benefit_gift.png',
            'url' => '/pages/activity/gift',
            'badge' => 0,
        ],
        [
            'id' => 2,
            'name' => '评价有礼',
            'sub_name' => '去评价',
            'icon' => '/static/images/benefit_review.png',
            'url' => '/pages/order/review',
            'badge' => $reviewCount,
        ],
        [
            'id' => 3,
            'name' => '品牌次卡',
            'sub_name' => '经典搭配',
            'icon' => '/static/images/benefit_card.png',
            'url' => '/pages/card/list',
            'badge' => 0,
        ],
    ];
}

/**
 * 获取等级配置
 * @param int $level
 * @return array
 */
private function getLevelConfig(int $level): array
{
    $configs = [
        0 => [
            'name' => 'V0 荷芽',
            'icon' => '🌱',
            'growth_value' => 0,
            'discount' => 1.0,
            'discount_text' => '无折扣',
        ],
        1 => [
            'name' => 'V1 荷叶',
            'icon' => '🌸',
            'growth_value' => 300,
            'discount' => 0.9,
            'discount_text' => '全场9折',
        ],
        2 => [
            'name' => 'V2 荷花',
            'icon' => '🌺',
            'growth_value' => 3000,
            'discount' => 0.85,
            'discount_text' => '全场85折',
        ],
    ];
    
    return $configs[$level] ?? $configs[0];
}

/**
 * 生成问候语
 * @return string
 */
private function getGreeting(): string
{
    $hour = intval(date('H'));
    
    if ($hour >= 6 && $hour < 12) {
        return '早上好！又是元气满满的一天';
    } elseif ($hour >= 12 && $hour < 18) {
        return '下午好！来杯下午茶放松一下';
    } elseif ($hour >= 18 && $hour < 22) {
        return '晚上好！辛苦一天了，放松一下吧';
    } else {
        return '夜深了，早点休息哦';
    }
}
```

<script>
export default {
  name: 'MemberCard',
  data() {
    return {
      memberInfo: null,
      bgImage: '/static/images/member-card-bg.jpg', // 会员卡背景图
    };
  },
  mounted() {
    this.loadMemberInfo();
  },
  methods: {
    // 加载会员信息
    async loadMemberInfo() {
      try {
        const res = await this.$api.get('/member/info');
        if (res.code === 200) {
          this.memberInfo = res.data;
        }
      } catch (e) {
        console.error('加载会员信息失败', e);
      }
    },
    
    // 获取下一等级名称
    getNextLevelName() {
      const levelNames = ['V0 荷芽', 'V1 荷叶', 'V2 荷花'];
      return levelNames[this.memberInfo.level + 1] || 'V2 荷花';
    },
    
    // 跳转预约页面
    goBooking() {
      uni.navigateTo({
        url: '/pages/booking/index'
      });
    },
  },
};
</script>
