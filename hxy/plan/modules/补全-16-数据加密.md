# 补全模块16：数据加密与脱敏

**优先级**：P3（可选）  
**影响范围**：数据安全、隐私保护、合规要求

## 一、问题描述

原方案未涉及**敏感数据加密和脱敏**，存在风险：

### 安全风险

1. **明文存储风险**
   - 用户手机号、身份证号明文存储
   - 数据库泄露后直接暴露

2. **日志泄露风险**
   - 接口日志记录完整手机号
   - 运维人员可直接查看

3. **前端展示风险**
   - 客服看到完整手机号
   - 技师看到用户完整信息

4. **合规要求**
   - 《个人信息保护法》要求加密存储
   - 《数据安全法》要求分级保护

---

## 二、解决方案

### 方案核心：存储加密 + 传输加密 + 展示脱敏 + 日志脱敏

## 四、加密方案设计

### 4.1 加密算法选择

| 场景 | 算法 | 密钥长度 | 说明 |
|------|------|----------|------|
| **存储加密** | AES-256-CBC | 256位 | 对称加密，性能好 |
| **传输加密** | HTTPS/TLS 1.3 | - | 传输层加密 |

### 4.2 加密工具类

```php
// app/utils/EncryptUtil.php

namespace app\utils;

class EncryptUtil
{
    // 从环境变量读取密钥
    private static function getKey()
    {
        return env('DATA_ENCRYPT_KEY', 'default-key-change-me');
    }
    
    private static function getIv()
    {
        return env('DATA_ENCRYPT_IV', '1234567890123456');
    }
    
    /**
     * AES-256-CBC 加密
     */
    public static function encrypt($data)
    {
        if (empty($data)) {
            return $data;
        }
        
        $key = self::getKey();
        $iv = self::getIv();
        
        $encrypted = openssl_encrypt(
            $data,
            'AES-256-CBC',
            $key,
            OPENSSL_RAW_DATA,
            $iv
        );
        
        return base64_encode($encrypted);
    }
    
    /**
     * AES-256-CBC 解密
     */
    public static function decrypt($data)
    {
        if (empty($data)) {
            return $data;
        }
        
        $key = self::getKey();
        $iv = self::getIv();
        
        $decrypted = openssl_decrypt(
            base64_decode($data),
            'AES-256-CBC',
            $key,
            OPENSSL_RAW_DATA,
            $iv
        );
        
        return $decrypted;
    }
    
    /**
     * 手机号脱敏
     */
    public static function maskPhone($phone)
    {
        if (empty($phone) || strlen($phone) != 11) {
            return $phone;
        }
        
        return substr($phone, 0, 3) . '****' . substr($phone, 7);
    }
    
    /**
     * 身份证号脱敏
     */
    public static function maskIdCard($idCard)
    {
        if (empty($idCard)) {
            return $idCard;
        }
        
        $len = strlen($idCard);
        if ($len == 18) {
            return substr($idCard, 0, 3) . '***********' . substr($idCard, 14);
        } elseif ($len == 15) {
            return substr($idCard, 0, 3) . '********' . substr($idCard, 11);
        }
        
        return $idCard;
    }
    
    /**
     * 姓名脱敏
     */
    public static function maskName($name)
    {
        if (empty($name)) {
            return $name;
        }
        
        $len = mb_strlen($name, 'UTF-8');
        if ($len == 1) {
            return $name;
        } elseif ($len == 2) {
            return mb_substr($name, 0, 1, 'UTF-8') . '*';
        } else {
            return mb_substr($name, 0, 1, 'UTF-8') . str_repeat('*', $len - 1);
        }
    }
    
    /**
     * 邮箱脱敏
     */
    public static function maskEmail($email)
    {
        if (empty($email) || !strpos($email, '@')) {
            return $email;
        }
        
        list($name, $domain) = explode('@', $email);
        $len = strlen($name);
        
        if ($len <= 2) {
            $masked = $name[0] . '*';
        } else {
            $masked = $name[0] . str_repeat('*', $len - 2) . $name[$len - 1];
        }
        
        return $masked . '@' . $domain;
    }
}
```

### 5.2 数据迁移脚本

```php
// app/command/EncryptData.php

namespace app\command;

use think\console\Command;
use think\console\Input;
use think\console\Output;
use think\facade\Db;
use app\utils\EncryptUtil;

class EncryptData extends Command
{
    protected function configure()
    {
        $this->setName('encrypt:data')
             ->setDescription('加密历史敏感数据');
    }
    
    protected function execute(Input $input, Output $output)
    {
        $output->writeln('开始加密历史数据...');
        
        // 1. 加密用户表
        $this->encryptUserData($output);
        
        // 2. 加密技师表
        $this->encryptTechnicianData($output);
        
        // 3. 加密订单表
        $this->encryptOrderData($output);
        
        $output->writeln('数据加密完成！');
    }
    
    private function encryptUserData($output)
    {
        $users = Db::name('user')
            ->where('phone_encrypted', '')
            ->where('phone', '<>', '')
            ->limit(1000)
            ->select();
        
        $count = 0;
        foreach ($users as $user) {
            Db::name('user')
                ->where('uid', $user['uid'])
                ->update([
                    'phone_encrypted' => EncryptUtil::encrypt($user['phone']),
                    'real_name_encrypted' => EncryptUtil::encrypt($user['real_name']),
                    'id_card_encrypted' => EncryptUtil::encrypt($user['id_card'])
                ]);
            $count++;
        }
        
        $output->writeln("用户表加密完成：{$count} 条");
    }
    
    private function encryptTechnicianData($output)
    {
        $technicians = Db::name('technician')
            ->where('phone_encrypted', '')
            ->where('phone', '<>', '')
            ->limit(1000)
            ->select();
        
        $count = 0;
        foreach ($technicians as $tech) {
            Db::name('technician')
                ->where('id', $tech['id'])
                ->update([
                    'phone_encrypted' => EncryptUtil::encrypt($tech['phone']),
                    'id_card_encrypted' => EncryptUtil::encrypt($tech['id_card'])
                ]);
            $count++;
        }
        
        $output->writeln("技师表加密完成：{$count} 条");
    }
    
    private function encryptOrderData($output)
    {
        $orders = Db::name('store_order')
            ->where('user_phone_encrypted', '')
            ->where('user_phone', '<>', '')
            ->limit(1000)
            ->select();
        
        $count = 0;
        foreach ($orders as $order) {
            Db::name('store_order')
                ->where('id', $order['id'])
                ->update([
                    'user_phone_encrypted' => EncryptUtil::encrypt($order['user_phone'])
                ]);
            $count++;
        }
        
        $output->writeln("订单表加密完成：{$count} 条");
    }
}
```

### 6.2 数据查询和展示

```php
// app/services/user/UserService.php

public function getUserInfo($userId, $needDecrypt = false)
{
    $user = Db::name('user')
        ->where('uid', $userId)
        ->find();
    
    if (!$user) {
        return null;
    }
    
    // 根据权限决定是否解密
    if ($needDecrypt) {
        // 管理员或用户本人可查看完整信息
        $user['phone'] = EncryptUtil::decrypt($user['phone_encrypted']);
        $user['real_name'] = EncryptUtil::decrypt($user['real_name_encrypted']);
        $user['id_card'] = EncryptUtil::decrypt($user['id_card_encrypted']);
    } else {
        // 其他人只能看脱敏信息
        $user['phone'] = EncryptUtil::maskPhone($user['phone']);
        $user['real_name'] = EncryptUtil::maskName($user['real_name']);
        $user['id_card'] = EncryptUtil::maskIdCard($user['id_card']);
    }
    
    // 移除加密字段
    unset($user['phone_encrypted']);
    unset($user['real_name_encrypted']);
    unset($user['id_card_encrypted']);
    
    return $user;
}
```

## 七、日志脱敏

### 7.1 日志中间件

```php
// app/middleware/LogMask.php

namespace app\middleware;

use Closure;
use app\utils\EncryptUtil;

class LogMask
{
    public function handle($request, Closure $next)
    {
        $response = $next($request);
        
        // 获取请求参数
        $params = $request->param();
        
        // 脱敏处理
        $maskedParams = $this->maskSensitiveData($params);
        
        // 记录日志
        trace('API请求：' . $request->url() . ' 参数：' . json_encode($maskedParams));
        
        return $response;
    }
    
    private function maskSensitiveData($data)
    {
        $sensitiveKeys = ['phone', 'id_card', 'real_name', 'password', 'bank_card'];
        
        foreach ($data as $key => &$value) {
            if (in_array($key, $sensitiveKeys)) {
                if ($key == 'phone') {
                    $value = EncryptUtil::maskPhone($value);
                } elseif ($key == 'id_card') {
                    $value = EncryptUtil::maskIdCard($value);
                } elseif ($key == 'real_name') {
                    $value = EncryptUtil::maskName($value);
                } else {
                    $value = '***';
                }
            } elseif (is_array($value)) {
                $value = $this->maskSensitiveData($value);
            }
        }
        
        return $data;
    }
}
```
