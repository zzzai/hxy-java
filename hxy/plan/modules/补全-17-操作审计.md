# 补全模块17：操作审计与权限细化

**优先级**：P3（可选）  
**影响范围**：安全审计、权限管理、合规要求  
**开发工作量**：1天

---

## 一、问题描述

原方案未涉及**操作审计和权限细化**，存在风险：

### 安全风险

1. **操作无法追溯**
   - 谁修改了商品价格？
   - 谁删除了订单？
   - 无法追查责任人

2. **权限过于粗放**
   - 店长能看所有数据
   - 客服能修改订单
   - 缺乏细粒度控制

3. **异常无法发现**
   - 批量修改价格
   - 异常退款操作
   - 缺乏预警机制

4. **合规要求**
   - 等保2.0要求操作审计
   - 金融行业要求完整日志

---

## 二、解决方案

### 方案核心：操作日志 + 权限细化 + 异常检测 + 审计报表

## 四、审计中间件实现

### 4.1 审计中间件

```php
// app/middleware/AuditLog.php

namespace app\middleware;

use Closure;
use think\facade\Db;
use think\facade\Request;

class AuditLog
{
    // 需要审计的路由
    private $auditRoutes = [
        'admin/product/save' => ['module' => 'product', 'action' => 'update'],
        'admin/order/refund' => ['module' => 'order', 'action' => 'refund'],
        'admin/user/update' => ['module' => 'user', 'action' => 'update'],
    ];
    
    public function handle($request, Closure $next)
    {
        $route = $request->rule()->getRule();
        
        // 判断是否需要审计
        if (!isset($this->auditRoutes[$route])) {
            return $next($request);
        }
        
        // 获取变更前数据
        $beforeData = $this->getBeforeData($request);
        
        // 执行请求
        $response = $next($request);
        
        // 获取变更后数据
        $afterData = $this->getAfterData($request);
        
        // 记录审计日志
        $this->logAudit(
            $this->auditRoutes[$route],
            $beforeData,
            $afterData,
            $request
        );
        
        return $response;
    }
    
    private function getBeforeData($request)
    {
        $id = $request->param('id');
        $module = $this->auditRoutes[$request->rule()->getRule()]['module'];
        
        if ($module == 'product') {
            return Db::name('store_product')->where('id', $id)->find();
        } elseif ($module == 'order') {
            return Db::name('store_order')->where('id', $id)->find();
        }
        
        return null;
    }
    
    private function getAfterData($request)
    {
        // 同上
    }
    
    private function logAudit($config, $before, $after, $request)
    {
        // 计算变更字段
        $changeFields = $this->getChangeFields($before, $after);
        
        // 计算风险等级
        $riskLevel = $this->calculateRiskLevel($config, $changeFields, $before, $after);
        
        // 插入日志
        Db::name('operation_log')->insert([
            'operator_id' => $request->adminId,
            'operator_type' => 1,
            'operator_name' => $request->adminInfo['real_name'] ?? '',
            'module' => $config['module'],
            'action' => $config['action'],
            'target_type' => $config['module'],
            'target_id' => $request->param('id'),
            'before_data' => json_encode($before),
            'after_data' => json_encode($after),
            'change_fields' => implode(',', $changeFields),
            'ip' => $request->ip(),
            'user_agent' => $request->header('user-agent'),
            'risk_level' => $riskLevel,
            'remark' => $this->generateRemark($config, $changeFields, $before, $after),
            'create_time' => time()
        ]);
    }
    
    private function getChangeFields($before, $after)
    {
        $changes = [];
        
        foreach ($after as $key => $value) {
            if (!isset($before[$key]) || $before[$key] != $value) {
                $changes[] = $key;
            }
        }
        
        return $changes;
    }
    
    private function calculateRiskLevel($config, $changeFields, $before, $after)
    {
        // 高风险操作
        if ($config['action'] == 'delete') {
            return 3;
        }
        
        // 价格变动超过50%
        if (in_array('price', $changeFields)) {
            $changeRate = abs($after['price'] - $before['price']) / $before['price'];
            if ($changeRate > 0.5) {
                return 2;
            }
        }
        
        // 退款金额超过1000
        if ($config['action'] == 'refund' && $after['refund_price'] > 1000) {
            return 2;
        }
        
        return 0;
    }
    
    private function generateRemark($config, $changeFields, $before, $after)
    {
        $remarks = [];
        
        foreach ($changeFields as $field) {
            $remarks[] = "{$field}: {$before[$field]} → {$after[$field]}";
        }
        
        return implode(', ', $remarks);
    }
}
```

### 5.3 权限检查

```php
// app/services/admin/PermissionService.php

namespace app\services\admin;

use think\facade\Db;

class PermissionService
{
    /**
     * 检查权限
     */
    public function checkPermission($adminId, $permissionCode)
    {
        // 1. 超级管理员跳过检查
        $admin = Db::name('system_admin')->where('id', $adminId)->find();
        if ($admin['level'] == 0) {
            return true;
        }
        
        // 2. 查询用户角色
        $roleIds = Db::name('system_admin_role')
            ->where('admin_id', $adminId)
            ->column('role_id');
        
        if (empty($roleIds)) {
            return false;
        }
        
        // 3. 查询角色权限
        $permissionIds = Db::name('admin_role_permission')
            ->whereIn('role_id', $roleIds)
            ->column('permission_id');
        
        if (empty($permissionIds)) {
            return false;
        }
        
        // 4. 检查权限代码
        $hasPermission = Db::name('admin_permission')
            ->whereIn('id', $permissionIds)
            ->where('code', $permissionCode)
            ->where('status', 1)
            ->count();
        
        return $hasPermission > 0;
    }
    
    /**
     * 获取用户权限列表
     */
    public function getUserPermissions($adminId)
    {
        $admin = Db::name('system_admin')->where('id', $adminId)->find();
        
        // 超级管理员返回所有权限
        if ($admin['level'] == 0) {
            return Db::name('admin_permission')
                ->where('status', 1)
                ->column('code');
        }
        
        // 普通用户返回角色权限
        $roleIds = Db::name('system_admin_role')
            ->where('admin_id', $adminId)
            ->column('role_id');
        
        $permissionIds = Db::name('admin_role_permission')
            ->whereIn('role_id', $roleIds)
            ->column('permission_id');
        
        return Db::name('admin_permission')
            ->whereIn('id', $permissionIds)
            ->where('status', 1)
            ->column('code');
    }
}
```

```php
// app/services/audit/AnomalyDetectionService.php

namespace app\services\audit;

use think\facade\Db;
use think\facade\Cache;

class AnomalyDetectionService
{
    /**
     * 检测批量操作
     */
    public function detectBatchOperation($adminId, $action)
    {
        $cacheKey = "batch_op:{$adminId}:{$action}";
        $count = Cache::get($cacheKey, 0);
        
        $count++;
        Cache::set($cacheKey, $count, 600);  // 10分钟
        
        if ($count > 10) {
            $this->sendAlert('批量操作异常', [
                'admin_id' => $adminId,
                'action' => $action,
                'count' => $count
            ]);
        }
    }
    
    /**
     * 检测价格异常
     */
    public function detectPriceAnomaly($productId, $oldPrice, $newPrice)
    {
        $changeRate = abs($newPrice - $oldPrice) / $oldPrice;
        
        if ($changeRate > 0.5) {
            $this->sendAlert('商品价格变动异常', [
                'product_id' => $productId,
                'old_price' => $oldPrice,
                'new_price' => $newPrice,
                'change_rate' => $changeRate
            ]);
        }
    }
    
    /**
     * 发送告警
     */
    private function sendAlert($message, $data)
    {
        // 1. 记录告警日志
        Db::name('audit_alert')->insert([
            'message' => $message,
            'data' => json_encode($data),
            'status' => 0,  // 待处理
            'create_time' => time()
        ]);
        
        // 2. 发送通知（企业微信/钉钉/邮件）

    }
}
```
