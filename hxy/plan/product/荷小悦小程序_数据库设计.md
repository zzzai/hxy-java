## 二、核心表结构详解

### 2.1 技师表 (eb_store_technician)

```sql
CREATE TABLE `eb_store_technician` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `store_id` int(11) unsigned NOT NULL DEFAULT '0' COMMENT '所属门店ID',
  `name` varchar(50) NOT NULL DEFAULT '' COMMENT '技师姓名',
  `work_no` varchar(20) NOT NULL DEFAULT '' COMMENT '工号',
  `avatar` varchar(255) NOT NULL DEFAULT '' COMMENT '头像',
  `phone` varchar(20) NOT NULL DEFAULT '' COMMENT '手机号',
  `password` varchar(255) NOT NULL DEFAULT '' COMMENT '登录密码',
  `specialty` varchar(500) DEFAULT NULL COMMENT '专长标签JSON ["肩颈调理", "祛湿艾灸"]',
  `intro` text COMMENT '个人简介',
  `work_years` tinyint(3) DEFAULT '0' COMMENT '从业年限',
  `certificates` varchar(500) DEFAULT NULL COMMENT '资质证书JSON',
  `rating` decimal(2,1) DEFAULT '5.0' COMMENT '评分 0-5',
  `rating_count` int(11) DEFAULT '0' COMMENT '评价数量',
  `order_count` int(11) DEFAULT '0' COMMENT '接单数量',
  `status` tinyint(1) DEFAULT '1' COMMENT '状态:0=离职 1=在职 2=休假',
  `is_busy` tinyint(1) DEFAULT '0' COMMENT '是否忙碌:0=空闲 1=忙碌',
  `sort` int(11) DEFAULT '0' COMMENT '排序',
  `create_time` int(11) DEFAULT '0',
  `update_time` int(11) DEFAULT '0',
  `delete_time` int(11) DEFAULT '0',
  PRIMARY KEY (`id`),
  KEY `store_id` (`store_id`),
  KEY `status` (`status`),
  KEY `is_busy` (`is_busy`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='技师表';
```

### 2.2 技师排班表 (eb_store_technician_schedule)

```sql
CREATE TABLE `eb_store_technician_schedule` (
  `id` int(11) unsigned NOT NULL AUTO_INCREMENT,
  `technician_id` int(11) unsigned NOT NULL COMMENT '技师ID',
  `store_id` int(11) unsigned NOT NULL COMMENT '门店ID',
  `work_date` date NOT NULL COMMENT '工作日期',
  `time_slot` varchar(20) NOT NULL COMMENT '时段 如:09:00-10:00',
  `start_time` time NOT NULL COMMENT '开始时间',
  `end_time` time NOT NULL COMMENT '结束时间',
  `is_available` tinyint(1) DEFAULT '1' COMMENT '是否可预约:0=否 1=是',
  `order_id` int(11) DEFAULT '0' COMMENT '占用订单ID',
  `create_time` int(11) DEFAULT '0',
  `update_time` int(11) DEFAULT '0',
  PRIMARY KEY (`id`),
  UNIQUE KEY `tech_date_slot` (`technician_id`, `work_date`, `time_slot`),
  KEY `store_date` (`store_id`, `work_date`),
  KEY `is_available` (`is_available`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='技师排班表';
```

### 2.3 订单表扩展字段 (eb_store_order)

```sql
-- 订单表新增预约相关字段
ALTER TABLE `eb_store_order`
ADD COLUMN `order_type` tinyint(1) NOT NULL DEFAULT '1' COMMENT '订单类型:1=商品 2=服务预约',
ADD COLUMN `store_id` int(11) NOT NULL DEFAULT '0' COMMENT '门店ID',
ADD COLUMN `technician_id` int(11) NOT NULL DEFAULT '0' COMMENT '预约技师ID',
ADD COLUMN `reserve_date` date DEFAULT NULL COMMENT '预约日期',
ADD COLUMN `reserve_time_slot` varchar(20) DEFAULT NULL COMMENT '预约时段',
ADD COLUMN `reserve_start_time` int(11) DEFAULT '0' COMMENT '预约开始时间戳',
ADD COLUMN `reserve_end_time` int(11) DEFAULT '0' COMMENT '预约结束时间戳',
ADD COLUMN `service_duration` int(11) NOT NULL DEFAULT '60' COMMENT '服务时长(分钟)',
ADD COLUMN `check_in_time` int(11) DEFAULT '0' COMMENT '到店核销时间',
ADD COLUMN `service_start_time` int(11) DEFAULT '0' COMMENT '服务开始时间',
ADD COLUMN `service_end_time` int(11) DEFAULT '0' COMMENT '服务结束时间',
ADD COLUMN `service_remark` text COMMENT '服务备注',
ADD COLUMN `technician_rating` tinyint(1) DEFAULT '0' COMMENT '技师评分',
ADD COLUMN `technician_comment` varchar(500) DEFAULT '' COMMENT '技师评价',
ADD COLUMN `shipping_type` tinyint(1) NOT NULL DEFAULT '2' COMMENT '配送方式:2=到店核销',
ADD COLUMN `gift_id` int(11) DEFAULT '0' COMMENT '礼品卡ID(如果是礼品兑换)';

-- 添加索引
ALTER TABLE `eb_store_order`
ADD INDEX `idx_order_type` (`order_type`),
ADD INDEX `idx_store_id` (`store_id`),
ADD INDEX `idx_technician_id` (`technician_id`),
```

## 五、数据迁移脚本

### 5.1 现有用户初始化会员等级

```sql
-- 根据累计消费初始化会员等级
UPDATE `eb_user` u
SET 
  `member_level` = CASE
    WHEN u.`pay_count` >= 10000 THEN 6
    WHEN u.`pay_count` >= 5000 THEN 5
    WHEN u.`pay_count` >= 2000 THEN 4
    WHEN u.`pay_count` >= 1000 THEN 3
    WHEN u.`pay_count` >= 500 THEN 2
    WHEN u.`pay_count` >= 100 THEN 1
    ELSE 0
  END,
  `growth_value` = COALESCE(u.`pay_count`, 0),
  `total_growth` = COALESCE(u.`pay_count`, 0);
```
