# 荷小悦O2O系统 - 架构修正方案（可上线版）

> **状态**：生产就绪 | **版本**：v2.0 | **基线**：database_migration_v2.0.sql
> 
> **修正原则**：事务一致性 > 幂等防重 > 租户隔离 > 平滑迁移

---

## 🔴 问题诊断

### 原方案致命缺陷

| 问题 | 级别 | 影响 | 证据 |
|------|------|------|------|
| 核销无事务/幂等 | 🔴 阻断 | 重复扣减、负余额、审计错账 | 示例代码未加锁/版本号 |
| 双账本冲突 | 🔴 阻断 | `eb_user_card_usage` vs `eb_card_usage` | v2.0已有bigint版本 |
| 加钟TOCTOU竞态 | 🟠 高 | 并发加钟同时通过 | 检测与确认间无锁 |
| 提成规则类型不匹配 | 🟠 高 | 规则静默失效 | "点钟"字符串 vs tinyint |
| 访问控制不完整 | 🟠 高 | 未校验角色/范围 | 只检查授权标志 |
| 无迁移回滚策略 | 🟠 高 | 生产切换风险 | 直接要求立即执行 |
| 租户隔离弱 | 🟡 中 | 数据串租风险 | 仅加字段无约束 |

```java
@Service
public class CardUsageService {
    
    @Transactional(rollbackFor = Exception.class)
    public CardUsageResult useCard(CardUsageRequest req) {
        // 1. 幂等检查
        String idempotentKey = req.getOrderId() + "_" + req.getUsageSeq();
        CardUsage existing = usageMapper.selectByIdempotentKey(idempotentKey);
        if (existing != null) {
            return CardUsageResult.success(existing); // 幂等返回
        }
        
        // 2. 乐观锁查询用户卡
        UserCard card = cardMapper.selectById(req.getUserCardId());
        if (card == null) {
            throw new BusinessException("卡不存在");
        }
        
        // 3. 业务校验
        if (card.getStatus() != 1) {
            throw new BusinessException("卡状态异常");
        }
        if (card.getBalance().compareTo(req.getUsageAmount()) < 0) {
            throw new BusinessException("余额不足");
        }
        
        // 4. 插入核销记录（带幂等键）
        CardUsage usage = new CardUsage()
            .setUserCardId(card.getId())
            .setUserId(card.getUserId())
            .setOrderId(req.getOrderId())
            .setUsageType(req.getUsageType())
            .setUsageAmount(req.getUsageAmount())
            .setBeforeAmount(card.getBalance())
            .setAfterAmount(card.getBalance().subtract(req.getUsageAmount()))
            .setTechnicianId(req.getTechnicianId())
            .setStoreId(req.getStoreId())
            .setIdempotentKey(idempotentKey)
            .setUsageSeq(req.getUsageSeq())
            .setStatus(1)
            .setCreatedAt((int)(System.currentTimeMillis() / 1000));
        
        try {
            usageMapper.insert(usage);
        } catch (DuplicateKeyException e) {
            // 并发插入，幂等返回
            return CardUsageResult.success(usageMapper.selectByIdempotentKey(idempotentKey));
        }
        
        // 5. 乐观锁更新卡余额
        int updated = cardMapper.updateBalanceWithVersion(
            card.getId(),
            card.getBalance().subtract(req.getUsageAmount()),
            card.getLockVersion()
        );
        
        if (updated == 0) {
            throw new BusinessException("卡余额更新失败，请重试"); // 触发事务回滚
        }
        
        return CardUsageResult.success(usage);
    }
}
```

```java
@Service
public class AddonService {
    
    @Autowired
    private RedissonClient redisson;
    
    public AddonResult handleAddon(AddonRequest req) {
        String lockKey = "addon_lock:" + req.getTechnicianId() + ":" + req.getTimeSlot();
        RLock lock = redisson.getLock(lockKey);
        
        try {
            // 1. 获取分布式锁（防止并发加钟）
            if (!lock.tryLock(3, 10, TimeUnit.SECONDS)) {
                throw new BusinessException("系统繁忙，请稍后重试");
            }
            
            // 2. 检测冲突（带版本号）
            List<Order> conflicts = orderMapper.selectConflicts(
                req.getTechnicianId(),
                req.getRoomId(),
                req.getStartTime(),
                req.getEndTime()
            );
            
            if (conflicts.isEmpty()) {
                // 无冲突，直接确认
                return confirmAddon(req);
            }
            
            // 3. 记录冲突（带快照）
            AddonConflict conflict = new AddonConflict()
                .setAddonOrderId(req.getOrderId())
                .setConflictOrderId(conflicts.get(0).getId())
                .setConflictType(detectType(req, conflicts.get(0)))
                .setConflictSnapshot(JSON.toJSONString(conflicts))
                .setCreatedAt((int)(System.currentTimeMillis() / 1000));
            
            conflictMapper.insert(conflict);
            
            // 4. 返回冲突，等待人工确认
            return AddonResult.conflict()
                .setConflictId(conflict.getId())
                .setConflicts(conflicts)
                .setOptions(Arrays.asList("换技师", "换房间", "顺延", "拒绝"));
            
        } finally {
            lock.unlock();
        }
    }
    
    @Transactional(rollbackFor = Exception.class)
    public AddonResult resolveConflict(Long conflictId, Integer resolution) {
        // 1. 乐观锁查询冲突记录
        AddonConflict conflict = conflictMapper.selectById(conflictId);
        if (conflict.getResolution() != 0) {
            throw new BusinessException("冲突已处理");
        }
        
        // 2. 根据决策执行
        switch (resolution) {
            case 3: // 顺延
                return postponeOrder(conflict);
            case 4: // 拒绝
                return rejectAddon(conflict);
            // ... 其他处理
        }
        
        // 3. 乐观锁更新冲突状态
        int updated = conflictMapper.updateResolutionWithVersion(
            conflictId,
            resolution,
            conflict.getLockVersion()
        );
        
        if (updated == 0) {
            throw new BusinessException("冲突状态已变更，请刷新");
        }
        
        return AddonResult.success();
    }
}
```

```java
@Service
public class CommissionEngine {
    
    @Transactional(rollbackFor = Exception.class)
    public BigDecimal calculate(Order order, Integer technicianId) {
        // 1. 获取生效规则
        List<CommissionRule> rules = ruleMapper.selectEffectiveRules(
            order.getStoreId(),
            order.getCreatedAt()
        );
        
        BigDecimal totalCommission = BigDecimal.ZERO;
        
        for (CommissionRule rule : rules) {
            if (!matchCondition(rule, order)) {
                continue;
            }
            
            // 2. 幂等检查
            String idempotentKey = order.getId() + "_" + technicianId + "_" + rule.getId();
            CommissionRecord existing = recordMapper.selectByIdempotentKey(idempotentKey);
            if (existing != null) {
                totalCommission = totalCommission.add(existing.getCommissionAmount());
                continue; // 幂等跳过
            }
            
            // 3. 计算提成
            BigDecimal amount = calculateByRule(rule, order);
            
            // 4. 插入记录（带幂等键）
            CommissionRecord record = new CommissionRecord()
                .setOrderId(order.getId())
                .setTechnicianId(technicianId)
                .setRuleId(rule.getId())
                .setRuleName(rule.getRuleName())
                .setBaseAmount(order.getTotalAmount())
                .setCommissionAmount(amount)
                .setIdempotentKey(idempotentKey)
                .setRecordNo(generateRecordNo())
                .setStatus(0)
                .setCreatedAt((int)(System.currentTimeMillis() / 1000));
            
            try {
                recordMapper.insert(record);
                totalCommission = totalCommission.add(amount);
            } catch (DuplicateKeyException e) {
                // 并发插入，幂等处理
                CommissionRecord concurrent = recordMapper.selectByIdempotentKey(idempotentKey);
                totalCommission = totalCommission.add(concurrent.getCommissionAmount());
            }
        }
        
        return totalCommission;
    }
    
    private boolean matchCondition(CommissionRule rule, Order order) {
        if (rule.getConditionExpr() == null) {
            return true;
        }
        
        JSONObject expr = JSON.parseObject(rule.getConditionExpr());
        JSONArray conditions = expr.getJSONArray("conditions");
        
        for (int i = 0; i < conditions.size(); i++) {
            JSONObject cond = conditions.getJSONObject(i);
            String field = cond.getString("field");
            String operator = cond.getString("operator");
            Object value = cond.get("value");
            
            // 类型安全的字段取值
            Object fieldValue = getFieldValue(order, field);
            
            if (!evaluate(fieldValue, operator, value)) {
                return false;
            }
        }
        
        return true;
    }
    
    private Object getFieldValue(Order order, String field) {
        switch (field) {
            case "dispatch_mode": return order.getDispatchMode(); // tinyint
            case "is_addon": return order.getIsAddon(); // tinyint
            case "order_amount": return order.getTotalAmount(); // decimal
            default: return null;
        }
    }
}
```

### 业务代码（完整访问控制）

```java
@Service
public class SensitiveDataService {
    
    @Autowired
    private AESEncryptor encryptor;
    
    @Autowired
    private AccessPermissionMapper permissionMapper;
    
    /**
     * 读取敏感数据（完整访问控制）
     */
    public String getSensitiveData(SensitiveDataRequest req) {
        // 1. 查询权限配置
        AccessPermission permission = permissionMapper.selectByRoleAndType(
            req.getAccessorType(),
            req.getDataType()
        );
        
        if (permission == null || permission.getCanRead() == 0) {
            // 记录拒绝日志
            logAccess(req, 3, "无权限");
            throw new BusinessException("无权访问该数据");
        }
        
        // 2. 检查是否需要填写原因
        if (permission.getRequireReason() == 1 && StringUtils.isEmpty(req.getReason())) {
            logAccess(req, 3, "未填写访问原因");
            throw new BusinessException("请填写访问原因");
        }
        
        // 3. 查询用户授权状态
        CustomerProfile profile = profileMapper.selectByUserId(req.getUserId());
        if (profile.getHealthAuthorized() == 0) {
            logAccess(req, 2, "用户未授权");
            throw new BusinessException("用户未授权访问健康信息");
        }
        
        // 4. 检查授权是否过期
        if (profile.getAuthorizationExpire() != null 
            && profile.getAuthorizationExpire().before(new Date())) {
            logAccess(req, 2, "授权已过期");
            throw new BusinessException("授权已过期，请重新申请");
        }
        
        // 5. 解密数据
        String decrypted = encryptor.decrypt(profile.getHealthEncrypted());
        
        // 6. 记录成功访问日志
        logAccess(req, 1, "访问成功");
        
        return decrypted;
    }
    
    /**
     * 记录访问日志（成功和失败都记录）
     */
    private void logAccess(SensitiveDataRequest req, int result, String remark) {
        SensitiveAccessLog log = new SensitiveAccessLog()
            .setUserId(req.getUserId())
            .setAccessorId(req.getAccessorId())
            .setAccessorType(req.getAccessorType())
            .setDataType(req.getDataType())
            .setDataField(req.getDataField())
            .setAccessReason(req.getReason())
            .setAccessResult(result)
            .setIpAddress(req.getIpAddress())
            .setUserAgent(req.getUserAgent())
            .setCreatedAt((int)(System.currentTimeMillis() / 1000));
        
        logMapper.insert(log);
    }
    
    /**
     * 保存敏感数据（加密）
     */
    @Transactional(rollbackFor = Exception.class)
    public void saveSensitiveData(Integer userId, String healthInfo, Integer operatorId) {
        // 1. 检查写权限
        AccessPermission permission = permissionMapper.selectByRoleAndType(
            getOperatorRole(operatorId),
            "health"
        );
        
        if (permission == null || permission.getCanWrite() == 0) {
            throw new BusinessException("无权修改该数据");
        }
        
        // 2. 加密存储
        String encrypted = encryptor.encrypt(healthInfo);
        
        CustomerProfile profile = profileMapper.selectByUserId(userId);
        profile.setHealthEncrypted(encrypted);
        profileMapper.updateById(profile);
        
        // 3. 记录操作日志
        SensitiveAccessLog log = new SensitiveAccessLog()
            .setUserId(userId)
            .setAccessorId(operatorId)
            .setAccessorType(getOperatorRole(operatorId))
            .setDataType("health")
            .setAccessReason("更新健康信息")
            .setAccessResult(1)
            .setCreatedAt((int)(System.currentTimeMillis() / 1000));
        
        logMapper.insert(log);
    }
}
```

```java
@Intercepts({

    @Signature(type = Executor.class, method = "update", args = {MappedStatement.class, Object.class})

public class TenantInterceptor implements Interceptor {
    
    @Override
    public Object intercept(Invocation invocation) throws Throwable {
        MappedStatement ms = (MappedStatement) invocation.getArgs()[0];
        Object parameter = invocation.getArgs()[1];
        
        // 1. 获取租户ID
        Integer tenantId = TenantContext.getTenantId();
        
        // 2. 解析SQL，注入租户条件
        BoundSql boundSql = ms.getBoundSql(parameter);
        String sql = boundSql.getSql();
        
        // 简化示例：实际需要用SQL解析器
        if (sql.contains("eb_user_card") || sql.contains("eb_card_usage")) {
            sql = injectTenantCondition(sql, tenantId);
        }
        
        // 3. 执行修改后的SQL
        return invocation.proceed();
    }
    
    private String injectTenantCondition(String sql, Integer tenantId) {
        // 实际实现需要用 JSqlParser 等工具

        if (sql.toLowerCase().contains("where")) {
            return sql.replaceFirst("(?i)where", "WHERE tenant_id = " + tenantId + " AND ");
        } else {
            return sql + " WHERE tenant_id = " + tenantId;
        }
    }
}
```
